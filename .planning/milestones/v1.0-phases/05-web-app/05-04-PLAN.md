Create `apps/web/src/hooks/use-optimistic-move.ts` that:
1. Tracks local board state (columns with items) for optimistic DnD updates
2. On `onDragOver`: uses `move()` from `@dnd-kit/helpers` to reorder items locally
3. On `onDragEnd`: calculates the fractional-index position string for the item at its new index, calls `useMoveItem` to persist to API
4. On API error: rolls back to the pre-drag state snapshot
5. Syncs with TanStack Query data when no drag is in progress

```typescript
'use client'

import { useState, useCallback, useRef, useEffect } from 'react'
import { move } from '@dnd-kit/helpers'
import type { Column, Item } from '@kanbambam/shared'
import { useMoveItem } from './use-api'
import { calculatePosition } from '@/lib/positions'

interface ColumnWithItems {
  id: string
  items: { id: string; position: string }[]
}

export function useOptimisticMove(
  columns: Column[],
  itemsByColumn: Record<string, Item[]>
) {
  const moveItem = useMoveItem()
  const [localItems, setLocalItems] = useState<Record<string, Item[]>>({})
  const isDragging = useRef(false)
  const preDropSnapshot = useRef<Record<string, Item[]>>({})

  // Sync from TanStack Query when not dragging
  useEffect(() => {
    if (!isDragging.current) {
      setLocalItems(itemsByColumn)
    }
  }, [itemsByColumn])

  // Current state: local during drag, TanStack Query otherwise
  const currentItems = Object.keys(localItems).length > 0 ? localItems : itemsByColumn

  const handleDragStart = useCallback(() => {
    isDragging.current = true
    preDropSnapshot.current = { ...currentItems }
  }, [currentItems])

  const handleDragOver = useCallback((event: any) => {
    // Use @dnd-kit/helpers move() for cross-list state updates
    setLocalItems((prev) => {
      // Build the nested structure move() expects
      const data: Record<string, { id: string; position: string }[]> = {}
      for (const col of columns) {
        data[col.id] = (prev[col.id] ?? itemsByColumn[col.id] ?? []).map(it => ({
          id: it.id,
          position: it.position,
        }))
      }
      const moved = move(data, event)
      // Map back to full Item objects
      const result: Record<string, Item[]> = {}
      for (const colId of Object.keys(moved)) {
        result[colId] = (moved[colId] as { id: string }[]).map(stub => {
          // Find the full item from any column
          for (const items of Object.values(prev.length ? prev : itemsByColumn)) {
            const found = (items as Item[]).find(it => it.id === stub.id)
            if (found) return found
          }
          // Fallback -- find across all columns
          for (const items of Object.values(itemsByColumn)) {
            const found = items.find(it => it.id === stub.id)
            if (found) return found
          }
          return stub as unknown as Item
        })
      }
      return result
    })
  }, [columns, itemsByColumn])

  const handleDragEnd = useCallback((event: any) => {
    isDragging.current = false

    if (event.canceled) {
      setLocalItems(preDropSnapshot.current)
      return
    }

    const { source, target } = event.operation
    if (!source || !target) return

    const itemId = source.id as string

    // Find which column the item is now in and at what index
    let targetColumnId: string | null = null
    let targetIndex = -1

    for (const col of columns) {
      const colItems = localItems[col.id] ?? []
      const idx = colItems.findIndex(it => it.id === itemId)
      if (idx !== -1) {
        targetColumnId = col.id
        targetIndex = idx
        break
      }
    }

    if (!targetColumnId || targetIndex === -1) return

    // Calculate fractional-index position
    const colItems = localItems[targetColumnId] ?? []
    const otherItems = colItems.filter(it => it.id !== itemId)
    const position = calculatePosition(otherItems, targetIndex)

    // Fire API call
    moveItem.mutate(
      { itemId, columnId: targetColumnId, position },
      {
        onError: () => {
          // Rollback to pre-drag state
          setLocalItems(preDropSnapshot.current)
        },
      }
    )
  }, [columns, localItems, moveItem])

  return {
    items: currentItems,
    handleDragStart,
    handleDragOver,
    handleDragEnd,
  }
}
```

NOTE: The `move()` helper from `@dnd-kit/helpers` may have a specific expected data shape. The executor should verify the actual API by reading the installed package types. The core pattern is: `move(currentState, event)` returns new state. If `move()` does not accept this shape, adapt by building the state structure it expects (typically a flat or nested array keyed by container/group ID).

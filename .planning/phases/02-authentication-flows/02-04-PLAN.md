export function output<T>(data: T, meta: Record<string, unknown>, options: {
	mode: OutputMode
	inkComponent: React.ReactElement
	plainText: string
}): void {
	switch (options.mode) {
		case 'json':
			outputJson(data, meta)
			break
		case 'plain':
			outputPlain(options.plainText)
			break
		case 'ink':
			outputInk(options.inkComponent)

	}
}

**1. Fix status.test.ts module caching bug:**

The test `--json outputs { data: { authenticated: true } } when logged in` fails because vitest caches the dynamically imported `status.ts` module. When the test sets `config.set('accessToken', 'test_access')` after the import, the cached module's `hasTokens()` still sees stale state from when the module was first loaded.

Fix: Add `vi.resetModules()` in the `beforeEach` hook to force fresh module imports for each test. This ensures each `await import('../status.js')` gets a clean module instance.

Update `apps/cli/src/commands/auth/__tests__/status.test.ts`:

Change the `beforeEach` block from:
```typescript
beforeEach(() => {
    config.clear()
})
```

To:
```typescript
beforeEach(() => {
    vi.resetModules()
    config.clear()
})
```

This ensures the dynamic `await import('../status.js')` in each test gets a fresh module, so `hasTokens()` correctly reads the config state set by that specific test.

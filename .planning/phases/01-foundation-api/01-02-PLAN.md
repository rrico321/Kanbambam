export const errorHandler: ErrorHandler = (err, c) => {

    const status = err.status
    return c.json({
      error: {
        code: err.message.toUpperCase().replace(/\s+/g, '_'),
        message: err.message,
        details: {},

- `signupRoute`: POST `/api/v1/auth/signup`, request body = `CreateUserSchema`, response 201 = envelope wrapping `AuthTokensSchema`, response 409 = `ErrorResponseSchema`
- `loginRoute`: POST `/api/v1/auth/login`, request body = `LoginSchema`, response 200 = envelope wrapping `AuthTokensSchema`, response 401 = `ErrorResponseSchema`
- `refreshRoute`: POST `/api/v1/auth/refresh`, request body = `z.object({ refreshToken: z.string() })`, response 200 = envelope wrapping `AuthTokensSchema`, response 401 = `ErrorResponseSchema`

2. Create `apps/api/src/routes/auth/auth.handlers.ts` with handler implementations:

**signup handler:**
- Validate input via Zod (automatic from OpenAPI route)
- Check if email already exists in `users` table. If yes, return 409 with `{ error: { code: "DUPLICATE_EMAIL", message: "Email already registered" } }`
- Hash password with `argon2.hash(password)`
- Generate user ID with `generateId('user')`
- Insert into `users` table
- Generate access token: use `jose` `SignJWT` with `{ sub: userId }`, algorithm `HS256`, secret from `env.JWT_SECRET`, expiry parsed from `env.JWT_EXPIRY` (e.g., "15m")
- Generate refresh token: random string via `nanoid(32)`, hash it with a simple SHA-256 (not argon2 -- too slow for refresh tokens), store hash in `refreshTokens` table with `generateId('refreshToken')` as ID, expiry parsed from `env.REFRESH_TOKEN_EXPIRY` (e.g., "7d")
- Return 201 with `{ data: { accessToken, refreshToken }, meta: { requestId } }`

**login handler:**
- Look up user by email. If not found, return 401 `{ error: { code: "INVALID_CREDENTIALS", message: "Invalid email or password" } }`
- Verify password with `argon2.verify(user.passwordHash, password)`. If mismatch, return same 401
- Generate access token and refresh token (same as signup)
- Return 200 with tokens

**refresh handler:**
- Receive `refreshToken` in body
- Hash the incoming token with SHA-256
- Look up matching hash in `refreshTokens` table. If not found or expired, return 401 `{ error: { code: "INVALID_REFRESH_TOKEN", message: "Invalid or expired refresh token" } }`
- Delete the used refresh token (rotation -- each token used once)
- Look up the user by `userId` from the refresh token record
- Generate new access token and new refresh token
- Return 200 with new tokens

**Token helper (create in same file or a separate `lib/tokens.ts`):**
```typescript
import { SignJWT } from 'jose'
import { createHash } from 'node:crypto'
import { nanoid } from 'nanoid'
import { env } from '../../env'

const secret = new TextEncoder().encode(env.JWT_SECRET)

export async function createAccessToken(userId: string): Promise<string> {
  return new SignJWT({ sub: userId })
    .setProtectedHeader({ alg: 'HS256' })
    .setIssuedAt()
    .setExpirationTime(env.JWT_EXPIRY)
    .sign(secret)
}

export function generateRefreshToken(): string {
  return nanoid(32)
}

export function hashToken(token: string): string {
  return createHash('sha256').update(token).digest('hex')
}
```
